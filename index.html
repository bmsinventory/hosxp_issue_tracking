▶ SQL สำหรับสร้าง Table ใน Supabase 
create table public.hospital_sheets (
  id            bigint generated always as identity primary key,
  hospital_name text   not null,
  product       text   not null,
  sheet_url     text   default '',
  gid           text   default '0',
  created_at    timestamptz default now(),
  unique(hospital_name, product)
);

alter table public.hospital_sheets enable row level security;

create policy "Allow all" on public.hospital_sheets
  for all using (true) with check (true);
